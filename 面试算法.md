// 简单工厂实现
interface DB{
	public function conn();
}


class MysqlDb implements DB{
	public function conn(){
		echo "链接到mysql";
	}
}


class RedisDb implements DB{
	public function conn(){
		echo "连接到Redis";
	}
}


class Mongodb implements DB{
	public function conn(){
		echo "连接到mongodb";
	}
}

class DB_factory{
	static $db_obj = null;
	public static function createDB($db_type){
		switch($db_type){
			case  'mysql':
			self::$db_obj = new MysqlDb();
			break;
			case  'redis':
			self::$db_obj = new RedisDb();
			break;
			case  'mongodb':
			self::$db_obj = new Mongodb();
			break;
		}
		return self::$db_obj;

	}
}

//链接mysql
$mysqldb = DB_factory::createDB('mysql');
$mysqldb->conn();

#----------------------------------------------------------------------------------------------

// 抽象工厂实现
interface DB{
	public function conn();
}


interface Factory{
	public function createdb();
}

// 实现DB的MySQL类
class mysqldb implements DB{
	public function conn(){
		echo "连接到MySQL";
	}
}

// 实现DB的Redis类
class redisdb implements DB{
	public function conn(){
		echo "连接到Redis";
	}

}

// 创建对应的数据库工厂类
class mysqlfactory implements Factory{
	public function createdb(){
		return new mysqldb();
	}
}

class redisfactory implements Factory{
	public  function createdb(){
		return new redisdb();
	}
}


// 测试

	}

	}
